import java.time.LocalDate;
import java.time.Period;
import java.util.*;
import java.util.function.Function;
import java.util.function.Predicate;
import java.util.stream.Stream;

public class GuviTask5_1 {
    //1) Write a program using map() method,to convert a list of Strings into uppercase.lf the
    //given List is: Stream names = Stream.of("aBc", "d", "ef');
    public static void main(String[] args) {
        Stream<String> stringStream=Stream.of("aBc", "d", "ef");
        Function<String,String> stringFunction=words->words.toUpperCase();
        System.out.println("Strings are changed to upperCase!");
        stringStream.map(stringFunction).forEach(result-> System.out.println(result));
    }
}
