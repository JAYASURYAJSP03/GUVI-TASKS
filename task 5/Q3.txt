import java.time.LocalDate;

import java.util.*;
import java.util.function.Function;
import java.util.function.Predicate;
import java.util.stream.Stream;

class  GuviTask5_3{
    //3.You are a teacher in school .1n your class there are 1O students , you have decided to give
    //special gifts to those students whose names start with "A".you are asked to separate those
    //students with the help of a java program .
    //Requirement:
    //Use List interface to store the student name
    //Use a lambda expression and the Stream API to filter the students
    public static void main(String[] args) {
        List<String > studentsName=new ArrayList<>(List.of("Aravind","Bala","Sooraj","Akash","Bharani","Madhavan","Andrew","pownraj","Ashwin","Akshaya"));
        Predicate<String> function=names-> names.startsWith("A");
        studentsName.forEach(names-> System.out.println("Is " +names+" starts with 'A' : "+function.test(names)));
        System.out.println("Students,Those who are selected for gifts!");
        studentsName.stream().filter(name->name.startsWith("A")).forEach(System.out::println);
    }
}
