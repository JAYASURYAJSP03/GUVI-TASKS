import java.util.Scanner;

public class Account {
    //1.3) Create Class Account with data member As Balance . Create two constructors (no argument,
    //and two arguments) and perform following task
    String accNo;
    static double balance;
    public Account(){
        balance=Math.random();
    }
    public  Account(String accNo,double balance){
        this.accNo=accNo;
        this.balance=balance;
    }
    public void deposit(int amount){
        //a.method to deposit the amount to the account .
        this.balance+=amount;
        System.out.println("your available balance after deposite is:"+balance);
    }
    public void withdraw(int amount){
        //b.method to withdraw the amount from the account .
        this.balance-=amount;
        System.out.println("your available balance after withdraw is:"+balance);
    }
    public void checkBalance(){
        //c.method to display the Balance .
        System.out.printf("your current available balance is %.2f\n",balance);
    }

    public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);
        System.out.println("Enter acc no : ");
        Account person=new Account(sc.nextLine(),Math.random());

        while(true){
            System.out.println("Menu (Enter options in only CAPITAL LETTERS)");
            System.out.println("1)CHECK BALANCE\n2)DEPOSITE AMOUNT\n3)WITHDRAW AMOUNT\n4)CLOSE");

            String option =sc.nextLine();
            if(option.equals("CHECK BALANCE")){
                person.checkBalance();
            }
            else if(option.equals("DEPOSITE AMOUNT")){
                System.out.println("Enter the amount to be deposite:");
                person.deposit(Integer.parseInt(sc.nextLine()));

            }
            else if(option.equals("WITHDRAW AMOUNT")){
                System.out.println("Enter the withdrawal amount:");
                person.withdraw(Integer.parseInt(sc.nextLine()));

            }
            else if(option.equals("CLOSE")) break;
            else{
                System.out.println("Enter the option correctly");
            }
        }
    }
}
