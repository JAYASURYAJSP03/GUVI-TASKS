import java.util.Scanner;

public class Person {
    //1.1)Create a class Person with properties (name and age) with following features.

    int age;
    String name;

    public Person(){
        age=18;//a. Default age of person should be 18;
    }

    public static void main(String[] args) {
        Person person=new Person();
        //b. A person object can be initialized with name and age;
        Scanner sc=new Scanner(System.in);
        System.out.println("Enter person's name and age :");

        person.name=sc.nextLine();
        person.age=sc.nextInt();
        person.display();
    }

    public void display(){
        //c. Method to display name and age of person
        System.out.println("Person name :"+name);
        System.out.println("Person age :"+age);
    }

}
