public class HelloWorld {
    //Write down the program to reverse the given number using loops .
    public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);
        int num=sc.nextInt();
        //Here i'm using while loop
        int res=0;
        while(num>0){
            res=(res*10)+(num%10);
            num/=10;
        }
        System.out.println(res);

    }
}
