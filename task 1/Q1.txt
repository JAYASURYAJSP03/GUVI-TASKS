public class HelloWorld {
    //for loop
    public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);
        int start=sc.nextInt();
        int end=sc.nextInt();
        for (int i = start; i <=end ; i++) {
            System.out.print(i);
        }


    }
}
public class HelloWorld {
    //while loop
    public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);
        int start=sc.nextInt();
        int end=sc.nextInt();
        while(start<=end){
            System.out.print(start+" ");
            start++;
        }

    }
}
